{{- if hugo.Store.Get "RENDERING_SLIDES" -}} {{- $slides := hugo.Store.Get "reveal_slides" | default (slice) -}} {{- $seen := hugo.Store.Get "reveal_seen" | default (dict) -}} {{- $body := .Page.RenderString (dict "display" "block" "renderShortcodes" true) .Inner -}} {{- $class := .Get "class" | default "" -}} {{- $attr := "" -}} {{- if $class -}} {{- $attr = printf ` class="%s"` $class -}} {{- end -}} {{- $html := printf "%s" $attr $body -}} {{- $sig := sha1 $html -}} {{- if not (index $seen $sig) -}} {{- $slides = $slides | append $html -}} {{- hugo.Store.Set "reveal_slides" $slides -}} {{- hugo.Store.Set "reveal_seen" (merge $seen (dict $sig true)) -}} {{- end -}} {{- end -}}
